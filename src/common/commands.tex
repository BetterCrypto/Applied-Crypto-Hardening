\def\idxnorm#1|{%

  \ifx\tempa\@empty%
    \textbar\relax%
  \else%
    \idx{#1}\xspace% 
  \fi%
}
\def|{\@ifnextchar[{\idxopt}{\idxnorm}%]
}
